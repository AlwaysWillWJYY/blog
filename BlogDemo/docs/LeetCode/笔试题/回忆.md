```java
        ArrayList<TreeNode> res = new ArrayList<>();
        if (pre_i > pre_j) {
            res.add(null);
            return res;
        }
        int rootVal = preOrder.get(pre_i);
        ArrayList<Integer> candidates = new ArrayList<>();
        for (int i = in_i; i <= in_j; i++) {
            if (inOrder.get(i) == rootVal) {
                candidates.add(i);
            }
        }

        if (candidates.size() != 0) {
            for (int i : candidates) {
                int leftTreeSize = i - in_i;
                ArrayList<TreeNode> leftTrees = buildTree(preOrder, inOrder, pre_i + 1, pre_i + leftTreeSize, in_i, i - 1);
                ArrayList<TreeNode> rightTrees = buildTree(preOrder, inOrder, pre_i + leftTreeSize + 1, pre_j, i + 1, in_j);
                for (TreeNode lefTree : leftTrees) {
                    for (TreeNode rightTree : rightTrees) {
                        TreeNode root = new TreeNode(rootVal);
                        root.left = lefTree;
                        root.right = rightTree;
                        res.add(root);
                    }
                }
            }
        }
        return res;
    }
```

```java
class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        StringBuilder sb = new StringBuilder();
        while(sc.hasNext()){
            sb.append(sc.next());
        }
        char[] a = String.join("",sb.toString()).toCharArray();
        Deque<Integer> de = new ArrayDeque<>();
        int n = a.length;
        int ans = 0;
        int cnt = 0;
        int isf = 0;

                de.addLast(isf);
                cnt += isf;
                isf = 0;
                i += 1;

                isf = 1;
                i += 3;
            }else if(a[i] == '}'){
                cnt -= de.pollLast();
                i += 1;
            }else{
                i += 1;
            }
            ans = Math.max(ans, cnt);
```

####  蚂蚁寻食的最短路径 

```java
import java.util.*;

public class Main {
    static int res = Integer.MAX_VALUE;
    public static void main(String[] args) {
        int[][] points = new int[3][2];
        points[0][0] = 0;
        points[0][1] = 1;
        points[1][0] = 0;
        points[1][1] = 2;
        points[2][0] = 0;
        points[2][1] = 3;
        System.out.println(getMinLen(points));
    }

    public static int getMinLen (int[][] points) {
        // write code here
        boolean[] visited = new boolean[points.length];
        backtrack(points,visited,0,0,0,0);
        return res;
    }
    public static void backtrack(int[][] points,boolean[] visited,int target,int length,int x,int y){
        if(target == points.length){
            res = Math.min(res,length);
            return;
        }
        for(int i = 0;i < points.length;i++){
            if(!visited[i]){
                length += Math.abs(points[i][0] - x) + Math.abs(points[i][1] - y);
                target++;
                visited[i] = true;
                backtrack(points,visited,target,length,points[i][0],points[i][1]);
                length -= Math.abs(points[i][0] - x) + Math.abs(points[i][1] - y);
                target--;
                visited[i] = false;
            }
        }
    }
}
```

```java
    static boolean vis[];
    static long ret;
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        ret = 0;
        list = new ArrayList<>();

            list.add(new ArrayList<>());
        }
        for(int i = 1; i < n; i++){
            int a = sc.nextInt();
            int b = sc.nextInt();
            list.get(a).add(b);
            list.get(b).add(a);
        }
        dfs(1, 1);

    }

    public static void dfs(int node, int del) {
        if(node < del || ret == -1) {
            ret = -1;
            return;
        }
        ret += node - del;
        vis[node] = true;
        for(int child : list.get(node)){
            if(!vis[child]){
                dfs(child, node);
            }
        }
    }
}
```

#### [最美子字符串的数目](https://leetcode.cn/problems/number-of-wonderful-substrings/)

```java
class Solution {
    public long wonderfulSubstrings(String word) {
        long[] cnt = new long[1 << 10];
        cnt[0] = 1; // 初始前缀和为 0，需将其计入出现次数
        long ans = 0L;
        for (int i = 0, sum = 0; i < word.length(); ++i) {
            sum ^= 1 << (word.charAt(i) - 'a'); // 计算当前前缀和
            ans += cnt[sum]; // 所有字母均出现偶数次
            for (int j = 0; j < 10; j ++) // 枚举其中一个字母出现奇数次
                ans += cnt[sum ^ (1 << j)]; // 反转该字母的出现次数的奇偶性
            ++cnt[sum]; // 更新前缀和出现次数
        }
        return ans;
    }
}
```

#### [买卖股票的最佳时机 IV](https://leetcode.cn/problems/best-time-to-buy-and-sell-stock-iv/)

```java
//股票k次交易
class Solution {
    public int maxProfit(int K, int[] prices) {
        int n = prices.length;
        if(prices.length == 0) return 0;
        //dp[i][j] ：第i天的状态为j，所剩下的最大现金是dp[i][j]
        int[][] dp = new int[n][2 * K + 1];
        for(int i = 1; i < 2 * K ; i+=2){
            dp[0][i] = -prices[0];
        }
        for(int i = 1; i < n; i++){
            for(int j = 0;j < 2 * K - 1; j += 2){
                //偶数是卖出，奇数是买入
                dp[i][j + 1] = Math.max(dp[i - 1][j + 1], dp[i - 1][j] - prices[i]);
                dp[i][j + 2] = Math.max(dp[i - 1][j + 2], dp[i - 1][j + 1] + prices[i]); 
            }
        }
        return dp[n - 1][2 * K];
    }
}
```

#### [从树中删除边的最小分数](https://leetcode.cn/problems/minimum-score-after-removals-on-a-tree/)

```java
class Solution {
    //先枚举删除一条边再dfs删除一条边
    int N = 1010;
    int[] h = new int[N];
    int[] e = new int[N * 2];
    int[] ne = new int[N * 2];
    int idx = 0;
    int[] w;
    int ans = Integer.MAX_VALUE;
    public void add(int a, int b){
        e[idx] = b;
        ne[idx] = h[a];
        h[a] = idx++;
    }
    
    public int dfs(int u, int fa, int sumx, int sumy){
        //返回的是异或值
        int res = w[u];
        for(int i = h[u]; i != -1; i = ne[i]){
            int j = e[i];
            if(j == fa) continue;//只向下走
            int t = dfs(j, u, sumx, sumy);
            res ^= t;
            //sumx, sumy, 
            if(sumx != -1){
                int[] a = new int[]{sumx ^ t, t, sumy};
                Arrays.sort(a);
                ans = Math.min(ans, a[2] - a[0]);
            }
        }
        return res;
    }
    public int minimumScore(int[] nums, int[][] edges) {
        w = nums;
        int n = nums.length;
        
        for(int i = 0;i < n - 1; i++){
            Arrays.fill(h, -1);//每次重新建图
            idx = 0;
            int a = edges[i][0];
            int b = edges[i][1];
            for(int j = 0; j < n - 1; j++){
                if(i != j){
                    int x = edges[j][0];
                    int y = edges[j][1];
                    add(x, y);
                    add(y, x);
                }
            }
            int sumx = dfs(a, -1, -1, -1);
            int sumy = dfs(b, -1, -1, -1);
            dfs(a, -1, sumx, sumy);
            dfs(b, -1, sumy, sumx);
        }
        return ans;
    }
}
```
